import Prose from "@/components/docs/Prose.astro";
import InstallTabs from "@/components/docs/InstallTabs.astro";

<div class="space-y-4">
<Prose>
### Initialize your project
</Prose>

<InstallTabs command="init" />

<Prose>
Sets up the `cn` utility, path aliases, and theme variables. Run this once before adding components.
</Prose>
</div>

<div class="space-y-4">
<Prose>
### Add components
</Prose>

<InstallTabs command="add button" />

<Prose>
Add multiple components at once:
</Prose>

<InstallTabs command="add button card dialog tabs" />

<Prose>
Add everything:
</Prose>

<InstallTabs command="add --all" />

<Prose>
Components are copied to `src/components/bearnie/`. Required utilities go to `src/utils/`.

Interactive components (like popover, command, and combobox) also install shared runtime files in `src/utils/runtime/` automatically. No additional loader setup is required.

You own the code.
</Prose>
</div>

<div class="space-y-4">
<Prose>
### List available components
</Prose>

<InstallTabs command="list" />

<Prose>
Shows all components grouped by category. Use `--json` for machine-readable output.
</Prose>
</div>

<div class="space-y-4">
<Prose>
### Options

| Flag | Description |
|------|-------------|
| `-y, --yes` | Skip prompts, overwrite existing files |
| `-a, --all` | Add all components |
| `--cwd <path>` | Set working directory |
| `--json` | Output as JSON (list command) |
</Prose>
</div>
